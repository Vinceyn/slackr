import datetime 

import threading 

import time 

 

def myTimer(time :datetime, executeFunction, functionArgs): 

now = datetime.datetime.now() 

sleepTime = (time-now).total_seconds() 

if sleepTime < 0: 

raise ValueError 

threading.Timer(sleepTime, executeFunction, functionArgs).start()
